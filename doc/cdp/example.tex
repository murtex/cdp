\section{Example}
\begin{itemize*}
	\item shown below is a minimal example of how to use framework's detection functions and plot corresponding tests
	\item audio data are supposed to be in file \code{'audio.wav'}, trial data in file \code{'log.txt'} and manually annotated data in file \code{'labels.xlsx'}
	\item application-specific functions \code{read\_audio}, \code{read\_trials} and \code{read\_labels} are supposed to read data properly into specified \code{run} structure
\end{itemize*}

\begin{tabular}{l}
	\hline
	\code{cfg = \textbf{cdf.hConfig}(); \color{gray}\textsl{\% default configuration}}\\
	\noalign{\medskip}
	\code{run = \textbf{cdf.hRun}(); \color{gray}\textsl{\% read raw data}}\\
	\code{\textbf{read\_audio}( run, 'audio.wav' );}\\
	\code{\textbf{read\_trials}( run, 'log.txt' );}\\
	\code{\textbf{read\_labels}( run, 'labels.xlsx' );}\\
	\noalign{\medskip}
	\code{offs = \textbf{cdf.sync}( run, cfg, false ); \color{gray}\textsl{\% sync timings}}\\
	\code{\textbf{cdf.plot.sync}( run, offs, 'sync.png' ); \color{gray}\textsl{\% plot test}}\\
	\noalign{\medskip}
	\code{\textbf{cdf.extract}( run, cfg ); \color{gray}\textsl{\% extract responses}}\\
	\code{detected = [run.trials.detected];}\\
	\code{detected = \textbf{cat}( 1, detected.range );}\\
	\code{labeled = [run.trials.labeled];}\\
	\code{labeled = \textbf{cat}( 1, labeled.range );}\\
	\code{\textbf{cdf.plot.extract}( run, detected, labeled, 'extract.png' ); \color{gray}\textsl{\% plot test}}\\
	\noalign{\medskip}
	\code{\textbf{cdf.landmark}( run, cfg ); \color{gray}\textsl{\% detect landmarks}}\\
	\code{detected = [run.trials.detected];}\\
	\code{detected = \textbf{cat}( 2, [detected.bo]', [detected.vo]', [detected.vr]' );}\\
	\code{labeled = [run.trials.labeled];}\\
	\code{labeled = \textbf{cat}( 2, [labeled.bo]', [labeled.vo]', [labeled.vr]' );}\\
	\code{\textbf{cdf.plot.landmark}( run, detected, labeled, 'landmark.png' ); \color{gray}\textsl{\% plot tests}}\\
	\code{\textbf{cdf.plot.timing}( run, detected, labeled, 'timing.png' );}\\
	\noalign{\medskip}
	\code{... \color{gray}\textsl{\% whatever needs to be done with detected data}}\\
	\hline
\end{tabular}
